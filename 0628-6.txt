// 연탄의 크기 (Softeer)

#include<iostream>
#include<algorithm>


using namespace std;

int n;
int arr[100];
int answer = 0;

int main(int argc, char** argv)
{
    ios::sync_with_stdio(false);
    cin.tie(0);

    
    
    cin >> n;
    for(int i=0; i<n; i++){
        cin >> arr[i];
    }

    sort(arr, arr+n);
    
    int maxVal = *max_element(arr, arr+n);
    
    for(int val=2; val<maxVal+1; val++){

        int cnt = 0;
        for(int i=0; i<n; i++){
            int target = arr[i];        
            if(target < val){
                continue;
            }
            if(target % val == 0){
                cnt++;
            }
        }

        answer = max(answer, cnt);
    }
    
    cout << answer;

   return 0;
}
